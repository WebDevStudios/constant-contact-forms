 53:      */
 54:     public function maybe_show_activation_message() {
 55: 
 56:         // If we have our query args where we're attempting to dismiss the notice
 57:         if (
 58:             isset( $_GET['ctct-dismiss'] ) &&
 59:             isset( $_GET['ctct-activation-action'] ) &&
 60:             $_GET['ctct-activation-action'] &&
 61:             wp_verify_nonce( $_GET['ctct-dismiss'], 'ctct-user-is-dismissing' )
 62:         ) {
 63: 
 64:             // Then save that we dismissed it
 65:             $this->save_dismissed_activation_message();
 66:         }
 67: 
 68:         // Only show if not connected & it wasn't dismissed
 69:         if ( ! $this->was_activation_message_dismissed() && ! constant_contact()->api->is_connected() ) {
 70:             $this->activation_message();
 71:         }
